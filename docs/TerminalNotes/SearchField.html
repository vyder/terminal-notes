def initialize(parent:)
    super(parent: parent, title: "Search",
          width: WIDTH, height: HEIGHT,
          x: POS_X, y: POS_Y,
          align: ALIGN)

    @text_field = TextField.new(@parent,
        position: @position.dup.moveBy(2, 2),
        width:    @width - 4)

    @has_focus = false

    draw
end
